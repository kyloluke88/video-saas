package video

// ========================
// 视觉风格（整体画风）
// ========================

type VisualStyle string

const (
	// 水墨国风，留白多，节奏慢，偏文化叙事
	VisualStyleInkWash VisualStyle = "ink_wash"

	// 绘本风格，柔和手绘质感，适合儿童故事
	VisualStyleStorybook VisualStyle = "storybook"

	// 日系动漫风格，色彩鲜明，角色表现力强
	VisualStyleAnime VisualStyle = "anime"

	// 扁平插画风格，色块清晰，信息表达明确
	VisualStyleFlat2D VisualStyle = "flat_2d"

	// 夸张卡通风格，适合搞笑或情绪强化
	VisualStyleCartoonExaggerated VisualStyle = "cartoon_exaggerated"

	// Q版可爱风，大头小身比例
	VisualStyleChibi VisualStyle = "chibi"

	// 半写实动画风格，比例自然，电影感强
	VisualStyleRealisticAnimation VisualStyle = "realistic_animation"

	// 3D卡通风格（类似商业动画）
	VisualStyleCartoon3D VisualStyle = "cartoon_3d"

	// 剪影风格，高对比抽象表达
	VisualStyleSilhouette VisualStyle = "silhouette"
)

// ========================
// 动画强度（运动节奏）
// ========================

type AnimationStyle string

const (
	// 静态镜头，无运动
	AnimationStyleStatic AnimationStyle = "static"

	// 轻微运动，缓慢推拉或轻微位移
	AnimationStyleSubtle AnimationStyle = "subtle"

	// 中等节奏，有明显动作但不过快
	AnimationStyleModerate AnimationStyle = "moderate"

	// 动态强烈，适合搞笑或冲突场景
	AnimationStyleDynamic AnimationStyle = "dynamic"

	// 快节奏剪辑型（不适合慢节奏寓言）
	AnimationStyleFastCut AnimationStyle = "fast_cut"
)

// ========================
// 表情夸张程度
// ========================

type ExpressionStyle string

const (
	// 真实自然，无夸张变形
	ExpressionStyleRealistic ExpressionStyle = "realistic"

	// 轻微夸张，适合儿童表达
	ExpressionStyleMildExaggerated ExpressionStyle = "mild_exaggerated"

	// 强烈夸张（星星绕头、眼睛变形等）
	ExpressionStyleFullExaggerated ExpressionStyle = "full_exaggerated"

	// 极简面部（情绪克制）
	ExpressionStyleMinimal ExpressionStyle = "minimal"
)

// ========================
// 镜头风格（可选扩展）
// ========================

type CameraStyle string

const (
	// 固定镜头
	CameraStyleStatic CameraStyle = "static"

	// 缓慢推镜
	CameraStyleSlowPush CameraStyle = "slow_push"

	// 轻微平移
	CameraStylePan CameraStyle = "pan"

	// 跟随角色
	CameraStyleFollow CameraStyle = "follow"

	// 俯视
	CameraStyleTopDown CameraStyle = "top_down"
)

# 关于镜头三层组合

✅ 一、镜头景别（Shot Size）
// ========================
// 镜头景别（画面远近）
// ========================

type ShotSize string

const (
	// 极远景，强调环境与氛围
	ShotExtremeWide ShotSize = "extreme_wide"

	// 远景，人物较小，环境为主
	ShotWide ShotSize = "wide"

	// 中远景
	ShotMediumWide ShotSize = "medium_wide"

	// 中景（人物半身）
	ShotMedium ShotSize = "medium"

	// 中近景（胸部以上）
	ShotMediumClose ShotSize = "medium_close"

	// 近景（面部为主）
	ShotCloseUp ShotSize = "close_up"

	// 特写（细节，例如眼睛、手）
	ShotExtremeClose ShotSize = "extreme_close"
)
✅ 二、镜头角度（Angle）
// ========================
// 镜头角度（拍摄视角）
// ========================

type CameraAngle string

const (
	// 平视（自然视角）
	AngleEyeLevel CameraAngle = "eye_level"

	// 轻微俯视（略有观察感）
	AngleHigh CameraAngle = "high"

	// 强俯视（上帝视角）
	AngleTopDown CameraAngle = "top_down"

	// 仰视（增强气势）
	AngleLow CameraAngle = "low"

	// 侧面构图
	AngleProfile CameraAngle = "profile"

	// 过肩镜头（对话常用）
	AngleOverShoulder CameraAngle = "over_shoulder"
)
